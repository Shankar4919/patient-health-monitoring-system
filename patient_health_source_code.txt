//ESP8266 Based Patient Health Monitoring System
#include <ESP8266WebServer.h>
#include <Wire.h>
#include <OneWire.h>
#include <DallasTemperature.h>
#include "DHT.h"
#include <PulseSensorPlayground.h> 

#define DHTTYPE DHT22
#define DHTPIN 14 //D5 pin= GPIO pin 14
#define DS18B20 2 //D4 pin= GPIO pin 2
#define REPORTING_PERIOD_MS     1000

//pulse pin
int PulseSensorPurplePin = 0;      
int Signal;            
int Threshold = 550; 

float temperature, humidity, BPM, bodytemperature;

/*Put your SSID & Password*/
const char* ssid = "mambafamily";  // Enter SSID here
const char* password = "mambafamily06";  //Enter Password here

DHT dht(DHTPIN, DHTTYPE);; //--> Initialize DHT sensor, DHT dht(Pin_used, Type_of_DHT_Sensor);
uint32_t tsLastReport = 0;
OneWire oneWire(DS18B20);
DallasTemperature sensors(&oneWire);

ESP8266WebServer server(80);

void setup() {
  Serial.begin(115200);
  pinMode(16, OUTPUT);
  delay(100);
  Serial.println(F("DHTxx test!"));
  dht.begin();
  Serial.println("Connecting to ");
  Serial.println(ssid);

  //connect to your local wi-fi network
  WiFi.begin(ssid, password);

  //check wi-fi is connected to wi-fi network
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println("WiFi connected..!");
  Serial.print("Got IP: ");  Serial.println(WiFi.localIP());

  server.on("/", handle_OnConnect);
  server.onNotFound(handle_NotFound);

  server.begin();
  Serial.println("HTTP server started");
  
}
void loop() {
  Signal = analogRead(PulseSensorPurplePin);
  server.handleClient();
  sensors.requestTemperatures();
  if(Signal > Threshold){                    
     digitalWrite(16,LOW);
   } else {
     digitalWrite(16,HIGH);      
   }
   delay(10);
  if (millis() - tsLastReport > REPORTING_PERIOD_MS) {
    float t = dht.readTemperature();
    String Temperature_Value = String(t);
    float h = dht.readHumidity();
    String Humidity_Value = String(h);
    bodytemperature = sensors.getTempCByIndex(0);
    
    temperature = t;
    humidity = h;
    BPM = Signal;


    Serial.print("Room Temperature: ");
    Serial.print(t);
    Serial.println("째C");

    Serial.print("Room Humidity: ");
    Serial.print(h);
    Serial.println("%");

    Serial.print("BPM: ");
    Serial.println(BPM);

    Serial.print("Body Temperature: ");
    Serial.print(bodytemperature);
    Serial.println("째C");

    Serial.println("*********************************");
    Serial.println();
    tsLastReport = millis();
  }


}

void handle_OnConnect() {

  server.send(200, "text/html", SendHTML(temperature, humidity, BPM, bodytemperature));
}

void handle_NotFound() {
  server.send(404, "text/plain", "Not found");
}
